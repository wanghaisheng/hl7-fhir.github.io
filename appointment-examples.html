5.14.5.1 Example

XML

General Person Example (id = "example")

Raw XML

<Appointment xmlns="http://hl7.org/fhir">
  <text>
    <status value="generated"/>
    <div xmlns="http://www.w3.org/1999/xhtml">Brian MRI results discussion</div>
  </text>
  <priority value="5">
  </priority>
  <status value="normal"/>
  <description value="Discussion on the results of your recent MRI"/>
  <start value="2013-12-10T09:00:00Z"/>
  <end value="2013-12-10T11:00:00Z"/>
  <location>
    <reference value="Location/1"/>
  </location>
  <comment value="Further expand on the results of the MRI and determine the next actions that may be appropriate."/>
  <participant>
    <individual>
      <reference value="Patient/example"/>
      <display value="Peter James Chalmers"/>
    </individual>
    <required value="required"/>
    <status value="accepted"/>
  </participant>
  <participant>
    <type>
      <coding>
        <code value="attending"/>
      </coding>
    </type>
    <individual>
      <reference value="Practitioner/example"/>
      <display value="Dr Adam Careful"/>
    </individual>
    <required value="required"/>
    <status value="accepted"/>
  </participant>
  <lastModifiedBy>
    <reference value="Practitioner/example"/>
      <display value="Dr Adam Careful"/>
  </lastModifiedBy>
  <lastModifiedDate value="2012-09-17"/>
</Appointment>
